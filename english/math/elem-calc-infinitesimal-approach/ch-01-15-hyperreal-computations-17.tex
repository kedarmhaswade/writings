\begin{hyperrule}[Reciprocals]
\begin{enumerate}
\item $\frac{1}{\varepsilon}$ Is Infinite.
\item $\frac{1}{b}$ Is Finite But Not Infinitesimal.
\item $\frac{1}{H}$ Is Infinitesimal.
\end{enumerate}
\end{hyperrule}
