We Assume The Following:
\begin{enumerate}
\item \alert{$\varepsilon,\delta$ Are Infinitesimal}.
\item \alert{$b,c$ Are Finite But Not Infinitesimal}.
\item \alert{$H,K$ Are Infinite}.
\end{enumerate}
\end{definition}
